Expand description

A sink that writes array chunks as an IPC stream.

The stream header is automatically written before writing the first chunk.

Examples

use futures::SinkExt;
use arrow2::array::{Array, Int32Array};
use arrow2::datatypes::{DataType, Field, Schema};
use arrow2::chunk::Chunk;
let schema = Schema::from(vec![
    Field::new("values", DataType::Int32, true),
]);

let mut buffer = vec![];
let mut sink = StreamSink::new(
    &mut buffer,
    &schema,
    None,
    Default::default(),
);

for i in 0..3 {
    let values = Int32Array::from(&[Some(i), None]);
    let chunk = Chunk::new(vec![values.boxed()]);
    sink.feed(chunk.into()).await?;
}
sink.close().await?;
